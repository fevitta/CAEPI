# CAEPI
Obter os dados do site do FTP do CAEPI, converter para UTF8 e inserir em uma base oracle

# Script Oracle
```
CREATE TABLE CAEPI (
  ID NUMBER,
  NRREGISTROCA VARCHAR2(50),
  DATAVALIDADE DATE,
  SITUACAO VARCHAR2(20),
  NRPROCESSO VARCHAR2(20),
  CNPJ VARCHAR2(20),
  RAZAOSOCIAL VARCHAR2(200),
  NATUREZA VARCHAR2(20),
  NOMEEQUIPAMENTO VARCHAR2(200),
  DESCRICAOEQUIPAMENTO VARCHAR2(4000),
  MARCACA VARCHAR2(100),
  REFERENCIA VARCHAR2(200),
  COR VARCHAR2(200),
  APROVADOPARALAUDO VARCHAR2(4000),
  RESTRICAOLAUDO VARCHAR2(4000),
  OBSERVACAOANALISELAUDO VARCHAR2(4000),
  CNPJLABORATORIO VARCHAR2(20),
  RAZAOSOCIALLABORATORIO VARCHAR2(200),
  NRLAUDO VARCHAR2(100),
  NORMA VARCHAR2(50),
  CONSTRAINT "CAEPI_PK" PRIMARY KEY ("ID")
)

CREATE INDEX CAEPI_NRREGISTROCA_IDX ON CAEPI (NRREGISTROCA);

CREATE SEQUENCE SEQ_CAEPI NOCACHE;

CREATE USER ROBO_GO
IDENTIFIED BY 123MUDAR;

GRANT CREATE SESSION TO ROBO_GO;
GRANT "CONNECT" TO ROBO_GO;
GRANT UNLIMITED TABLESPACE TO ROBO_GO;
GRANT INSERT, UPDATE, DELETE, SELECT ON CAEPI TO ROBO_GO;
GRANT DROP ANY TABLE TO ROBO_GO;
GRANT SELECT ON SEQ_CAEPI TO ROBO_GO;
```
